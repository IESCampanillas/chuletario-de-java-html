int cifra = numInt;
int cogNum = 0;
int contDigit = 0;
    
for ( int h = 1; h <= numInt; h *= 10 ) {
	contDigit++;
}
    
for ( int i = 1; i <= contDigit; i++ ) {
  int potencia10 = (int)(Math.pow(10,(contDigit-i)));
  cogNum = cifra / potencia10;
  System.out.print(cogNum+"\n");
  cifra = cifra - (cogNum*potencia10);
}
